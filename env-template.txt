# Copy this file to .env.local and fill in your Spotify app credentials

NEXTAUTH_URL=http://localhost:3000
NEXTAUTH_SECRET=your-secret-key-here-generate-a-random-string
SPOTIFY_CLIENT_ID=your-spotify-client-id-from-spotify-developer-dashboard
SPOTIFY_CLIENT_SECRET=your-spotify-client-secret-from-spotify-developer-dashboard

# To get Spotify credentials:
# 1. Go to https://developer.spotify.com/dashboard
# 2. Create a new app
# 3. Set redirect URI to: http://localhost:3000/api/auth/callback/spotify
# 4. Copy Client ID and Client Secret to this file
